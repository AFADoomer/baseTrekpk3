Actor TOSPlantonRedAlert : TOSUsableSpriteDecoration 20217
{
	//$Title "Wall Planton: RedAlert" 
	Height 16
	States
	{
		Spawn:
			PO01 A 1 NoDelay A_SpawnItemEx("TOSRedAlertLight", 120.0, 0.0, 0.0)
		Static:
			PO01 A 1 A_JumpIf(CallACS("A_CheckAlert") > 0, "Alert")
			Loop
		Alert:
			PO01 B 35 A_JumpIf(CallACS("A_CheckAlert") < 1, "Static")
			PO01 A 35 A_JumpIf(CallACS("A_CheckAlert") < 1, "Static")
			Loop
	}
}

Actor TOSPlantonIntercom : TOSUsableSpriteDecoration 20224
{
	//$Title "Wall Planton: Intercom (Large)"
	Height 18
	Radius 8
	Activation "THINGSPEC_Switch"
	States
	{
		Spawn:
			PO08 A 1 NoDelay A_SpawnItemEx("TOSRedAlertLightSmall", 32.0, 0.0, 18.0)
			Goto Static
		Static:
			PO08 # 0 A_StopSound(CHAN_AUTO)
			Goto StaticLoop
		StaticLoop:
			PO08 # 1 A_JumpIf(CallACS("A_CheckAlert") > 0, "Alert")
			Loop
		Alert:
			PO08 # 0 A_PlaySound("ambient/redalert", CHAN_AUTO, 1.0, 1)
			PO08 B 35
			PO08 A 35 A_JumpIf(CallACS("A_CheckAlert") < 1, "Static")
			Loop
		Active:
		Inactive:
			PO01 # 0 A_PlaySound("intercom", CHAN_AUTO, 1.0)
			Goto StaticLoop
	}
}
